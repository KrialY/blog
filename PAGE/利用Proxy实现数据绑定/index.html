    利用Proxy实现数据绑定：

    对象：

    // object
let obj = {
  a:1,
  b:2
}
let proxy = new Proxy(obj, {
  get (target, prop) {
    console.log("my proxy:" + target[prop]);
    return target[prop];
  },
  set (target, prop, value) {
    console.log("My proxy change data:" + target[prop]);
    target[prop] = value;
  }
});
console.log(obj);
console.log(proxy.a);
console.log(obj);
proxy.a=3;
console.log(obj);
let arr = [1,2,3,4,5];
