    class MyPromise {
  constructor (executor) {
    this.state="pending";
    this.value=undefined;
    this.reason=undefined;
    this.onResolvedCb = [];
    this.onRejectedCb = [];

    const resolve = (value) => {
      if(this.state === "pending"){
        this.state = "fulfilled";
        this.value = value;
      }
      this.onResolvedCb.forEach(fn => fn());
    }
    const reject = (reason) => {
      if(this.state === "pending"){
        this.state = "rejected";
        this.reason = reason;
      }
      this.onRejectedCb.forEach(fn => fn());
    }
    executor(resolve,reject);
  }
  then (resolve, reject) {
    const promise2 = new MyPromise((rs, rj) => {
      if (this.state === "fulfilled") {
        setTimeout(() => { // 这里加setTimeout利用了event loop机制，这里是微任务
          const x = resolve(this.value);
          resolvePromise(promise2, x, rs, rj);
        },0);
      }else if (this.state === "rejected") {
        setTimeout(() => {
          const x = reject(this.reason);
          resolvePromise(promise2, x, rs, rj);
        },0);
      }else if (this.state === "pending") {
        this.onResolvedCb.push(() => {
          setTimeout(() => {
            const x = resolve(this.value);
            resolvePromise(promise2, x, rs, rj);
          },0);
        });
        this.onRejectedCb.push(() => {
          setTimeout(() => {
            const x = reject(this.reason);
            resolvePromise(promise2, x, rs, rj);
          },0);
        });
      }
    });
    return promise2; 
  }
}
