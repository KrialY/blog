var maxSlidingWindow = function(nums, k) {
    if(k<=0) return [];
    let res=new Array();
    for(let i=0;i<=nums.length-k;i++){
        let max=nums[i];
        for(let j=i+1;j<i+k;j++){
            if(nums[j]>max){
                max=nums[j];
            }
        }
        res.push(max);
    }
    return res;
};
